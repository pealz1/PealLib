Quick Start

Get up and running in under a minute

1

Load the library

local repo = 'https://raw.githubusercontent.com/pealz1/PealLib/main/'

local Library      = loadstring(game:HttpGet(repo .. 'Library.lua'))()
local ThemeManager = loadstring(game:HttpGet(repo .. 'addons/ThemeManager.lua'))()
local SaveManager  = loadstring(game:HttpGet(repo .. 'addons/SaveManager.lua'))()
2

Create a window and add elements

local Window = Library:CreateWindow({
    Title    = 'My Script Hub',
    Center   = true,
    AutoShow = true
})

local Tab = Window:AddTab('Main')
local Box = Tab:AddLeftGroupbox('Features')

Box:AddToggle('MyToggle', {
    Text    = 'Enable Feature',
    Default = false,
    Callback = function(value)
        print('Toggled:', value)
    end
})

Box:AddSlider('MySlider', {
    Text     = 'Walk Speed',
    Default  = 16,
    Min      = 0,
    Max      = 100,
    Rounding = 0,
    Suffix   = ' studs/s'
})
3

Add themes and config saving

local SettingsTab = Window:AddTab('Settings')

ThemeManager:SetLibrary(Library)
SaveManager:SetLibrary(Library)
SaveManager:IgnoreThemeSettings()

ThemeManager:ApplyToTab(SettingsTab)
SaveManager:BuildConfigSection(SettingsTab)

-- Auto-load saved config on start
SaveManager:LoadAutoloadConfig()

UI Elements

Complete reference for every available element

Toggle

Boolean on/off switch with optional keybind attachment.

Box:AddToggle('Index', {
    Text     = 'Label',
    Default  = false,
    Tooltip  = 'Description',
    Callback = function(v) end
})

-- Read value globally
Toggles.Index.Value  -- true / false
Text Default Tooltip Callback

Slider

Numeric range slider with configurable min, max, rounding, and suffix.

Box:AddSlider('Index', {
    Text     = 'Label',
    Default  = 50,
    Min      = 0,
    Max      = 100,
    Rounding = 0,
    Suffix   = '%',
    Compact  = false,
    HideMax  = false
})
Text Default Min Max Rounding Suffix Compact HideMax

Dropdown

Single or multi-select dropdown. Supports dynamic player lists.

-- Single select
Box:AddDropdown('Index', {
    Text   = 'Label',
    Values = {'A', 'B', 'C'},
    Default = 1
})

-- Multi select
Box:AddDropdown('Index', {
    Values = {'Red', 'Green', 'Blue'},
    Multi  = true
})

-- Player list
Box:AddDropdown('Target', {
    SpecialType = 'Player'
})
Values Default Multi SpecialType AllowNull Text

Input

Text input field with optional numeric-only mode and max length.

Box:AddInput('Index', {
    Text        = 'Label',
    Default     = '',
    Placeholder = 'Type here...',
    Numeric     = false,
    Finished    = false,
    MaxLength   = 64
})
Text Default Placeholder Numeric Finished MaxLength

Color Picker

RGB/HSV color selector with optional transparency slider.

-- Attach to a label
Box:AddLabel('Color'):AddColorPicker('Index', {
    Default      = Color3.fromRGB(255, 0, 0),
    Title        = 'Pick Color',
    Transparency = 0
})

Options.Index.Value         -- Color3
Options.Index.Transparency  -- 0-1
Default Title Transparency Callback

Keybind Picker

Configurable keybind with Always, Toggle, and Hold modes.

-- Attach to a label or toggle
Box:AddLabel('Trigger'):AddKeyPicker('Index', {
    Default  = 'MB2',
    Mode     = 'Toggle',
    Text     = 'Trigger Key'
})

Options.Index.Value  -- key name
Options.Index.Mode   -- mode string
Default Mode Text NoUI SyncToggleState

Button

Clickable button with sub-button nesting and double-click support.

local btn = Box:AddButton({
    Text        = 'Execute',
    DoubleClick = true,
    Tooltip     = 'Double-click to confirm',
    Func = function()
        print('Executed!')
    end
})

-- Nest a sub-button
btn:AddButton({
    Text = 'Cancel',
    Func = function() end
})
Text Func Tooltip DoubleClick

Dependency Box

Container that shows/hides its children based on other element states.

local depbox = Box:AddDependencyBox()

depbox:SetupDependencies({
    { Toggles.MyToggle, true }
})

-- Elements inside only show when
-- MyToggle is true
depbox:AddSlider('DepSlider', {
    Text = 'Conditional',
    Default = 50, Min = 0,
    Max = 100, Rounding = 0
})

Other Elements

Box:AddLabel(text, doesWrap) Static or dynamic text label
Box:AddDivider() Horizontal separator line
Box:AddBlank(size) Vertical spacing
Box:AddRow(config) Horizontal button layout
Box:AddCard(config) Info card with badge and buttons

API Reference

Core library methods and global tables

Library Methods

Library:CreateWindow(config) Create the main application window
Library:CreatePopout(config) Create a floating auxiliary window
Library:Notify(message, duration) Show a notification toast
Library:SetWatermark(text) Set watermark text content
Library:SetWatermarkVisibility(bool) Toggle watermark display
Library:Unload() Destroy all UI and clean up
Library:OnUnload(callback) Register a cleanup callback
Library:AddToolTip(text, instance) Add hover tooltip to any instance
Library:MakeDraggable(instance) Enable drag-and-drop on a frame

Window Methods

Window:AddTab(name) Add a new tab, returns Tab object

Tab Methods

Tab:AddLeftGroupbox(name) Add a groupbox to the left column
Tab:AddRightGroupbox(name) Add a groupbox to the right column
Tab:AddLeftTabbox() Add a tabbox container to the left
Tab:AddRightTabbox() Add a tabbox container to the right

Global Tables

Toggles.Index.Value Access any toggle's boolean state
Options.Index.Value Access any element's current value (sliders, dropdowns, inputs, etc.)

Element Methods

Element:OnChanged(callback) Listen for value changes after creation
Element:SetValue(value) Programmatically update the element's value
Dropdown:SetValues(array) Replace the dropdown's option list
ColorPicker:SetValueRGB(color3, transparency) Set color picker value programmatically
KeyPicker:GetState() Check if the keybind is currently active

ThemeManager

ThemeManager:SetLibrary(Library)
ThemeManager:SetFolder('MyScript')

-- Apply to a settings tab
ThemeManager:ApplyToTab(SettingsTab)

-- Or apply to a specific groupbox
ThemeManager:ApplyToGroupbox(MyGroupbox)

-- Programmatic theme switching
ThemeManager:ApplyTheme('Tokyo Night')

SaveManager

SaveManager:SetLibrary(Library)
SaveManager:SetFolder('MyScript')
SaveManager:IgnoreThemeSettings()

-- Add config UI to a tab
SaveManager:BuildConfigSection(SettingsTab)

-- Auto-load on start
SaveManager:LoadAutoloadConfig()

-- Programmatic save/load
SaveManager:Save('my_config')
SaveManager:Load('my_config')
